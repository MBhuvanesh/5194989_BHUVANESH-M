#include <stdio.h>
#include <string.h>
#include <ctype.h>

char* pangrams(char* s) {
    int seen[26] = {0};   
    int unique = 0;

    for (int i = 0; s[i] != '\0'; i++) {
        if (isalpha(s[i])) {
            int idx = tolower(s[i]) - 'a';
            if (seen[idx] == 0) {
                seen[idx] = 1;
                unique++;
            }
        }
    }

    if (unique == 26)
        return "pangram";
    else
        return "not pangram";
}
